
CREATE TABLE tbl_installment (
		inst_id	int NOT NULL AUTO_INCREMENT, 
		inst_loan_amount double,
		inst_interest_rate double,
		inst_period_month int,
		inst_first_installment_date date,
		inst_date date,
		inst_num int,
		inst_amt_pay double,
		inst_amt_principle double,
		inst_amt_interest double,
		inst_amt_balance double,
		inst_amt_penalty double,
		inst_cancel_yn varchar(1),
		sell_id int,
		regDt	datetime,
		upDt	datetime,
		useYn	varchar(1),
		regUsr	int,
		upUsr	int,
		com_id	int,
		PRIMARY KEY(inst_id) 
);

CREATE TABLE tbl_installment_payment (
		inst_paid_id int NOT NULL AUTO_INCREMENT, 
		inst_total_paid_amount double,
		inst_paid_yn varchar(1),
		inst_paid_date date,
		inst_paid_des text,
		inst_paid_status varchar(1),
		inst_cancel_yn varchar(1),
		rec_id int,
		met_id int,
		inst_id int,
		regDt	datetime,
		upDt	datetime,
		useYn	varchar(1),
		regUsr	int,
		upUsr	int,
		com_id	int,
		PRIMARY KEY(inst_paid_id) 
);
